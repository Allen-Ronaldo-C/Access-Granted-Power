#include <Keypad.h>
#include <SimpleDHT.h>

// Password Setup
const char* password = "427";  
int position = 0;

// Keypad Setup
const byte ROWS = 4;
const byte COLS = 4;
char keys[ROWS][COLS] = {
  {'1','2','3','A'},
  {'4','5','6','B'},
  {'7','8','9','C'},
  {'*','0','#','D'}
};
byte rowPins[ROWS] = {8, 7, 6, 9};
byte colPins[COLS] = {5, 4, 3, 2};
Keypad keypad = Keypad(makeKeymap(keys), rowPins, colPins, ROWS, COLS);

// Lock Control Pins
int RedpinLock = 12;
int GreenpinUnlock = 13;
int relayPin = 10;

// Sensor Setup
#define DHTPIN A0        
SimpleDHT11 dht11(DHTPIN);  

#define MQ2Pin A1       
// Thresholds
float tempThreshold = 40.0;
int gasThreshold = 300;      

void setup() {
  Serial.begin(9600);  
  pinMode(RedpinLock, OUTPUT);
  pinMode(GreenpinUnlock, OUTPUT);
  pinMode(relayPin, OUTPUT);
  
  LockedPosition(true);  

  Serial.println("System Ready. Enter 3-digit password:");
}

void loop() {
  monitorEnvironment(); 
  
  char key = keypad.getKey();

  if (key) {
    Serial.print("Key Pressed: ");
    Serial.println(key);
  }

  if (key == '*' || key == '#') {
    position = 0;
    LockedPosition(true);
    Serial.println("Reset. Try again.");
  }

  if (key == password[position]) {
    position++;
    Serial.print("Correct digit ");
    Serial.print(position);
    Serial.println("/3");
  } else if (key && key != password[position]) {
    position = 0;
    Serial.println("Wrong digit. Resetting...");
    LockedPosition(true);
  }

  if (position == 3) {
    Serial.println("Access Granted. Power ON.");
    LockedPosition(false);
  }

  delay(100);
}

void LockedPosition(int locked) {
  if (locked) {
    digitalWrite(RedpinLock, HIGH);
    digitalWrite(GreenpinUnlock, LOW);
    digitalWrite(relayPin, LOW);
  } else {
    digitalWrite(RedpinLock, LOW);
    digitalWrite(GreenpinUnlock, HIGH);
    digitalWrite(relayPin, HIGH);
  }
}

void monitorEnvironment() {
  byte temperature = 0;
  byte humidity = 0;

  int err = dht11.read(&temperature, &humidity, NULL);

  if (err != SimpleDHTErrSuccess) {
    Serial.print("DHT11 Read failed, err="); Serial.println(err);
    return;
  }

  int gasValue = analogRead(MQ2Pin);

  Serial.print("Temp: ");
  Serial.print((int)temperature);
  Serial.print(" Â°C, Humidity: ");
  Serial.print((int)humidity);
  Serial.print(" %, Gas: ");
  Serial.println(gasValue);

  if (temperature > tempThreshold) {
    Serial.println("ALERT: High Temperature!");
    digitalWrite(RedpinLock, HIGH);
    digitalWrite(GreenpinUnlock, LOW);
    digitalWrite(relayPin, LOW);
  }

  if (gasValue > gasThreshold) {
    Serial.println("ALERT: Gas Leakage Detected!");
    digitalWrite(RedpinLock, HIGH);
    digitalWrite(GreenpinUnlock, LOW);
    digitalWrite(relayPin, LOW);
  }
}
